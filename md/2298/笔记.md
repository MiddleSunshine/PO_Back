```php
<?php
class C {
	public function __construct(public string $name='ccccc')
	{
	}
}
class B {
	public function __construct(public C $c)
	{
	}
}
class A {
	public function __construct(public B $b)
	{
	}
}

$a = new A(new B(new C));
// 如要访问c的name属性
/**
 * php 7的写法
 */

if (!is_null($a)) {
	if (!is_null($a->b)) {
		if (!is_null($a->b->c)) {
			echo $a->b->c->name;
		}
	}
}
/**
 * php 8的写法
 */
echo $a?->b?->c?->name;
?>
```
