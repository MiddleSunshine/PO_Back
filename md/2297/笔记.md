```php
<?php

$data = "1";

switch ($data) {
    case 1:
        echo "int 1";
        break;
    case "1":
        echo "string 1";
        break;
    default:
        echo "I don't know";
}
echo PHP_EOL;

$result = match ($data) {
    "1" => "string 1",
    1 => "int 1",
    default => "I don't know"
};

echo $result;
```
